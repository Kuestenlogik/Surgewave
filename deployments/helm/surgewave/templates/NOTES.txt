== Quick Start ==

  # Produce a test message
  kubectl run surgewave-test --rm -it --image=surgewave:latest --restart=Never -- \
    surgewave-cli produce --bootstrap {{ $fullname }}:{{ .Values.broker.ports.kafka }} --topic test-topic --message "hello surgewave"

  # Consume messages
  kubectl run surgewave-test --rm -it --image=surgewave:latest --restart=Never -- \
    surgewave-cli consume --bootstrap {{ $fullname }}:{{ .Values.broker.ports.kafka }} --topic test-topic --from-beginning

== Scaling ==

  # Scale brokers
  kubectl scale statefulset {{ $fullname }} --replicas=5 -n {{ .Release.Namespace }}

  # Or use Helm
  helm upgrade {{ .Release.Name }} deploy/helm/surgewave --set broker.replicaCount=5
